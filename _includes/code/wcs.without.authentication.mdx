<TabItem value="go" label="Go">

```go
package main

import (
	"context"
	"fmt"
	"github.com/weaviate/weaviate-go-client/v4/weaviate"
)

func main() {
    cfg := weaviate.Config{
        Host:   "some-endpoint.weaviate.network",  // Replace with your endpoint
        Scheme: "https",
    }

    client, err := weaviate.NewClient(cfg)
    if err != nil {
      panic(err)
    }
}
```
