import json

client = weaviate.Client('https://some-endpoint.semi.network')
query_result = client.query.get(class_name='Publication', properties="name").with_near_vector({
    "vector": [-0.14, -0.06, -0.07],

weaviate = require("weaviate-client");

const client = weaviate.client({
    scheme: "https",
    host: "some-endpoint.semi.network"
});

client.graphql
    .get()
    .withClassName('Publication')
    .withFields('name _additional {vector certainty}')
    .withNearVector({"vector": [-0.14, -0.06, -0.07], "certainty": 0.7})
    .do()
    .then(res => {
        console.log(res)
    })
    .catch(err => {
        console.error(err)
    });
