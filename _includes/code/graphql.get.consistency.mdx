import weaviate
from weaviate.data.replication import ConsistencyLevel

client = weaviate.Client("http://localhost:8080")

resp = (
    client.query.get("Article", ["name"])
    .with_additional("isConsistent")
    .with_consistency_level(ConsistencyLevel.ONE)
    .do()
)

print(f"resp: {resp}")
