```python
from uuid import uuid4

generate_prompt = """
Please write an engaging advertisement for the following AirBnb Listing:
Description: {description}
"""

listing_collection = client.collections.use("Listing")
ad_collection = client.collections.use("Ad")

response = listing_collection.generate.fetch_objects(
    limit=5,
    single_prompt=generate_prompt,
)

for ad in response.objects:
    new_ad_id = uuid4()
    ad_collection.data.insert(
        properties={"content": ad.generated},
        uuid=new_ad_id,
    )
    listing_collection.data.reference_add(
        from_uuid=ad.uuid,
        from_property="hasAd",
        to=new_ad_id,
    )
```

```ts
const generatePrompt: string = 'Please write an engaging advertisement for the following AirBnB Listing: Description: {description}';

const listings = client.collections.use('Listing');
const ads = client.collections.use('Ad');

const response = await listings.generate.fetchObjects(
  { singlePrompt: generatePrompt },
  { limit: 5 }
);

for (const ad of response.objects) {
  const newAdId = await ads.data.insert({
    properties: { content: ad.generative?.text },
  });
  await listings.data.referenceAdd({
    fromUuid: ad.uuid,
    fromProperty: 'hasAd',
    to: newAdId,
  });
}
```
