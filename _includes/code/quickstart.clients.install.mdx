import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs groupId="languages">
<TabItem value="py" label="Python">

Add `weaviate-client` to your Python environment with `pip`:<br/><br/>

```bash
$ pip install weaviate-client
```

</TabItem>
<TabItem value="js" label="JavaScript/TypeScript">

Add `weaviate-ts-client` to your project with `npm`:<br/><br/>

```bash
npm install weaviate-ts-client
```

</TabItem>
<TabItem value="go" label="Go">

Add `weaviate-go-client` to your project with `go get`:<br/><br/>

```bash
go get github.com/weaviate/weaviate-go-client/v4
```

</TabItem>
<TabItem value="java" label="Java">

Add this dependency to your project:<br/><br/>

```xml
<dependency>
  <groupId>io.weaviate</groupId>
  <artifactId>client</artifactId>
  <version>4.0.0</version>  <!-- Check latest version -->
</dependency>
```

</TabItem>
